【Git構築】
git init .
git remote add origin https://github.com/miyoshiakira/ai-image-creater.git
git add .
git commit -m '初回コミット'
git push origin master

【環境構築】
npm create vite@latest ai-image-creater -- --template react-ts
cd ai-image-creater
npm install
npm run dev

【Hosting & Storage & Function(Python)】
npm install -g firebase-tools
firebase login
firebase init
